\section{Normalización de las labels}

\begin{frame}{Normalización de las Labels}
    \begin{columns}[T]
        \column{0.4\textwidth}
        
        \begin{block}{Por qué normalizar}
            Las variables tienen escalas muy distintas: $R \sim 20$ \, px pero  $P \sim 0.3$. Sin normalizar, las variables grandes dominan el MAE y el modelo aprende mal.
        \end{block}

        \begin{exampleblock}{StandardScaler}
            Técnica que transforma las variables:
            $$x_{norm} = \frac{x - \mu}{\sigma}$$
            donde $\mu$ es media y $\sigma$ es desviación estándar.
            
            \textbf{Resultado:} media 0 y desv. est. 1.
        \end{exampleblock}
        
        \column{0.45\textwidth}
        
        \begin{exampleblock}{Transformación Inversa}
            Despejar $x$ de la ecuación:
            $$x = x_{norm} \cdot \sigma + \mu$$
            
            \textbf{Importante:} Los valores de $\mu$ y $\sigma$ se guardan.
        \end{exampleblock}
        
        \begin{block}{Flujo del proceso}
            \small
            \begin{enumerate}
                \item CSV Original con 5 variables
                \item Aplicar StandardScaler
                \item Guardar scaler
                \item Labels listas para entrenar
                \item Entrenar la red neuronal
                \item Usar Scaler Inverso para obtener MAPE
            \end{enumerate}
        \end{block}
    \end{columns}
\end{frame}

\section{Pipeline de Entrenamiento}

\begin{frame}{Transformación de Imágenes}
    \begin{block}{Valores originales}
        Las imágenes SEM vienen con píxeles en rango $[0, 255]$.
    \end{block}
    
    \begin{exampleblock}{Normalización de píxeles}
        Se transforman a uno de estos rangos:
        \begin{itemize}
            \item Rango $[0, 1]$: dividir por 255
            \item Rango $[-1, 1]$: normalizar a media 0 y desv. est. 1
            \item Mantener $[0, 255]$: sin transformación
        \end{itemize}
    \end{exampleblock}
    
    \begin{alertblock}{Por qué}
        Facilita convergencia del modelo y evita dominancia numérica de valores grandes.
    \end{alertblock}
\end{frame}

\begin{frame}{División del Dataset}
    \begin{block}{Partición}
        Las 20\,000 imágenes se dividen en:
        \begin{itemize}
            \item \textbf{Train Set}: entrena el modelo
            \item \textbf{Validation Set}: valida en cada época
            \item \textbf{Test Set}: evalúa al final
        \end{itemize}
    \end{block}
    
    \begin{block}{Batching}
        Las imágenes se agrupan en lotes para:
        \begin{itemize}
            \item Optimizar memoria GPU/CPU
            \item Acelerar entrenamiento
            \item Mejorar estabilidad numérica
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Entrenamiento del Modelo}
    \begin{block}{Arquitectura}
        Dos opciones:
        \begin{itemize}
            \item \textbf{CNN Completa}: red convolucional entrenada desde cero
            \item \textbf{Transfer Learning}: últimas capas convolucionales de modelo preentrenado
        \end{itemize}
    \end{block}
    
    \begin{block}{Datos de entrada}
        \begin{itemize}
            \item Imágenes: transformadas a $[0,1]$, $[-1,1]$ o $[0,255]$
            \item Labels: normalizados con StandardScaler (guardado en \texttt{scaler.pkl})
        \end{itemize}
    \end{block}
    
    \begin{block}{Proceso}
        Minimizar pérdida en Train Set validando en cada época.
    \end{block}
\end{frame}

\begin{frame}{Obtención del MAPE en Test Set}
    \begin{block}{Paso 1: Predicciones}
        El modelo entrenado realiza predicciones en Test Set:
        $$ \text{Modelo}(x_{imagen}) = (\hat{R}_{norm}, \hat{\sigma_R}_{norm}, \hat{DCC}_{norm}, \hat{\sigma_{DCC}}_{norm}, \hat{P}_{norm}) $$
    \end{block}
    
    \begin{block}{Paso 2: Scaler Inverso}
        Recuperar labels al espacio original usando \texttt{scaler.pkl}:
        $$(\hat{R}, \hat{\sigma_R}, \hat{DCC}, \hat{\sigma_{DCC}}, \hat{P}) = \text{scaler.inverse\_transform}(\hat{R}_{norm}, \hat{\sigma_R}_{norm}, \ldots)$$
    \end{block}
    
    \begin{exampleblock}{Paso 3: Calcular MAPE}
        $$\text{MAPE} = \frac{1}{n}\sum_{i=1}^{n}\left|\frac{y_i - \hat{y}_i}{y_i}\right| \times 100\text{\%}$$
        donde $y_i$ son los labels reales del Test Set.
    \end{exampleblock}
\end{frame}

\begin{frame}{Evaluación en Dataset Externo Real}
    \begin{columns}[T]
        \column{0.5\textwidth}
        
        \begin{alertblock}{Datos externos}
            Dataset de imágenes SEM \textbf{reales} (no simuladas) para evaluar generalización.
        \end{alertblock}
        
        \begin{block}{Función: evaluate\_external\_dataset}
            \begin{enumerate}
                \item Cargar imágenes SEM reales
                \item Aplicar misma transformación de píxeles
                \item Realizar predicciones con modelo entrenado
                \item Aplicar Scaler Inverso (\texttt{scaler.pkl})
                \item Calcular MAPE
            \end{enumerate}
        \end{block}
        
        \begin{exampleblock}{Resultado}
            MAPE en datos reales indica rendimiento en datos no vistos.
